class KNNClassifier():

  def fit(self, X, y):
    self.X=X
    self.y=y

  def predict(self, X, K, epsilon =1e-3):

    N=len(X)
    y_hat = np.zeros(N)
    for i in range(N):
      dist2 = np.sum((self.X-X[i])**2, axis=1)
      idxt  = np.argsort(dist2)[:K]
      gamma_k = 1/(np.sqrt(dist2[idxt]+epsilon))
      y_hat[i] = np.bincount(self.y[idxt], weights= gamma_k).argmax()

    return y_hat

def smote(X, y, k=5, oversampling_ratio=1.0, random_seed=None, minority_class = None):

    if random_seed:
        np.random.seed(random_seed)
    if minority_class == None:
      minority_class = np.unique(y)[np.argmin(np.bincount(y))]
    minority_indices = np.where(y == minority_class)[0]
    synthetic_samples = int(len(minority_indices) * oversampling_ratio)
    print(synthetic_samples)
    X_resampled = X.copy()
    y_resampled = y.copy()

    for i in range(synthetic_samples):
        random_index = np.random.choice(minority_indices)
        target_sample = X[random_index]

        # Find k-nearest neighbors
        distances = np.linalg.norm(X - target_sample, axis=1)
        sorted_indices = np.argsort(distances)
        nearest_neighbors = sorted_indices[1:k+1]

        # Choose one of the nearest neighbors randomly
        selected_neighbor = X[np.random.choice(nearest_neighbors)]

        # Generate synthetic sample
        alpha = np.random.rand()
        synthetic_sample = target_sample + alpha * (selected_neighbor - target_sample)

        # Append synthetic sample to the resampled data
        X_resampled = np.vstack([X_resampled, synthetic_sample])
        y_resampled = np.append(y_resampled, minority_class)

    return X_resampled, y_resampled

def OLS(Y,Y_hat,N): #capital Y becoz we gonna have multiple targets , like having vector Y
  return (1/(2*N)*np.sum(Y-Y_hat)**2)     #target function -> J=summation (y-y_hat)**2

def R2(Y,Y_hat):
  return (1-(np.sum((Y-Y_hat)**2)/np.sum((Y-np.mean(Y))**2)))

class MVLinearRegression():

  def fit(self, X, y, eta=1e-3, epochs=1e3, show_curve=False):   #epochs is how many time we want to run all steps of SGD
    epochs = int(epochs)
    N, D = X.shape
    Y = y

    #begin optimization
    self.W = np.random.randn(D)   #random initialize
    J = np.zeros(epochs)

    #stochastig Gradient Descent
    for epoch in range(epochs):
      Y_hat = self.predict(X)
      J[epoch] = OLS(Y,Y_hat,N)   #we are going to get values here

      #weight update rule
      self.W -= eta* (1/N)*(X.T@(Y_hat-Y))    #This is eta*gradient

    if show_curve:
      plt.figure()
      plt.plot(J)
      plt.xlabel("epochs")
      plt.ylabel("$\mathcal{J}$")
      plt.title("Training Curve")
      plt.show()

  def predict(self, X):
    return X@self.W     #data times weights
